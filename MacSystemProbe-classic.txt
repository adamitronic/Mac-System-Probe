PROGRAM MacSystemProbe;CONST  kPanelOverview = 1;  kPanelMemory   = 2;  kPanelDisplay  = 3;  kPanelRaw      = 4;  kPanelAbout    = 5;  kFirstPanel    = kPanelOverview;  kLastPanel     = kPanelAbout;  kPanelCount    = 5;  kAppleMenuID   = 128;  kFileMenuID    = 129;  kViewMenuID    = 130;  kMenuBarID     = 200;  kAboutAlertID  = 256;  kCmdRefresh    = 1;  kCmdQuit       = 2;  kLineHeight    = 12;  kMaxLines      = 96;  gestaltMachineType     = 'mach';  gestaltProcessorType   = 'proc';  gestaltKeyboardType    = 'kbd ';  gestaltSystemVersion   = 'sysv';  gestaltQuickdrawVersion = 'qd  ';  gestaltLogicalRAMSize  = 'lram';  gestaltPhysicalRAMSize = 'ram ';  gestaltROMSize         = 'rom ';  gestaltROMVersion      = 'romv';  gestaltMMUType         = 'mmu ';  gestaltFPUType         = 'fpu ';TYPE  LineText = STRING[127];  LineArray = ARRAY[0..95] OF LineText;  TProbeInfo = RECORD    machineType: INTEGER;    systemVersion: INTEGER;    processor: INTEGER;    hasFPU: BOOLEAN;    hasColorQD: BOOLEAN;    keyBoardType: INTEGER;    sysVRefNum: INTEGER;  END;VAR  gDone: BOOLEAN;  gMainWindow: WindowPtr;  gAppleMenu: MenuHandle;  gFileMenu: MenuHandle;  gViewMenu: MenuHandle;  gNavButtons: ARRAY[kFirstPanel..kLastPanel] OF ControlHandle;  gScrollBar: ControlHandle;  gCurrentPanel: INTEGER;  gTopLine: INTEGER;  gVisibleLines: INTEGER;  gLineCount: INTEGER;  gLines: LineArray;  gSysEnv: TProbeInfo;  gGestaltAvailable: BOOLEAN;  gContentRect: Rect;FUNCTION MinI(a, b: INTEGER): INTEGER;BEGIN  IF a < b THEN    MinI := a  ELSE    MinI := b;END;FUNCTION MaxI(a, b: INTEGER): INTEGER;BEGIN  IF a > b THEN    MaxI := a  ELSE    MaxI := b;END;FUNCTION LongToStr(value: LONGINT): Str255;VAR  s: Str255;BEGIN  NumToString(value, s);  LongToStr := s;END;FUNCTION HexDigit(value: INTEGER): CHAR;BEGIN  IF value < 10 THEN    HexDigit := CHR(ORD('0') + value)  ELSE    HexDigit := CHR(ORD('A') + value - 10);END;FUNCTION HexWord(value: LONGINT): Str255;VAR  s: Str255;  temp: LONGINT;BEGIN  temp := value MOD 65536;  IF temp < 0 THEN    temp := temp + 65536;  s := '$0000';  s[2] := HexDigit((temp DIV 4096) MOD 16);  s[3] := HexDigit((temp DIV 256) MOD 16);  s[4] := HexDigit((temp DIV 16) MOD 16);  s[5] := HexDigit(temp MOD 16);  HexWord := s;END;FUNCTION HexLong(value: LONGINT): Str255;VAR  s: Str255;  temp: LONGINT;BEGIN  temp := value;  s := '$00000000';  s[2] := HexDigit((temp DIV 268435456) MOD 16);  s[3] := HexDigit((temp DIV 16777216) MOD 16);  s[4] := HexDigit((temp DIV 1048576) MOD 16);  s[5] := HexDigit((temp DIV 65536) MOD 16);  s[6] := HexDigit((temp DIV 4096) MOD 16);  s[7] := HexDigit((temp DIV 256) MOD 16);  s[8] := HexDigit((temp DIV 16) MOD 16);  s[9] := HexDigit(temp MOD 16);  HexLong := s;END;FUNCTION BoolToStr(flag: BOOLEAN): Str255;BEGIN  IF flag THEN    BoolToStr := 'Yes'  ELSE    BoolToStr := 'No';END;FUNCTION Plural(unitName: Str255; amount: LONGINT): Str255;BEGIN  IF amount = 1 THEN    Plural := unitName  ELSE    Plural := CONCAT(unitName, 's');END;FUNCTION KBytes(bytes: LONGINT): Str255;VAR  whole: LONGINT;BEGIN  whole := bytes DIV 1024;  KBytes := CONCAT(LongToStr(whole), ' KB');END;FUNCTION MBytes(bytes: LONGINT): Str255;VAR  whole: LONGINT;BEGIN  whole := bytes DIV (1024 * 1024);  IF whole > 0 THEN    MBytes := CONCAT(LongToStr(whole), ' MB')  ELSE    MBytes := KBytes(bytes);END;FUNCTION VersionWordToStr(versionWord: LONGINT): Str255;VAR  majorPart: LONGINT;  minorPart: LONGINT;  bugPart: LONGINT;  s: Str255;BEGIN  versionWord := versionWord MOD 65536;  IF versionWord < 0 THEN    versionWord := versionWord + 65536;  majorPart := (versionWord DIV 256) MOD 256;  minorPart := (versionWord DIV 16) MOD 16;  bugPart := versionWord MOD 16;  s := LongToStr(majorPart);  s := CONCAT(s, '.');  s := CONCAT(s, LongToStr(minorPart));  IF bugPart <> 0 THEN  BEGIN    s := CONCAT(s, '.');    s := CONCAT(s, LongToStr(bugPart));  END;  VersionWordToStr := s;END;FUNCTION RectToStr(r: Rect): Str255;VAR  s: Str255;BEGIN  s := LongToStr(r.left);  s := CONCAT(s, ',');  s := CONCAT(s, LongToStr(r.top));  s := CONCAT(s, ' to ');  s := CONCAT(s, LongToStr(r.right));  s := CONCAT(s, ',');  s := CONCAT(s, LongToStr(r.bottom));  RectToStr := s;END;FUNCTION ProcessorName(procID: LONGINT): Str255;BEGIN  CASE procID OF    1: ProcessorName := 'Motorola 68000';    2: ProcessorName := 'Motorola 68010';    3: ProcessorName := 'Motorola 68020';    4: ProcessorName := 'Motorola 68030';    5: ProcessorName := 'Motorola 68040';  OTHERWISE    ProcessorName := CONCAT('Unknown CPU (', CONCAT(LongToStr(procID), ')'));  END;END;FUNCTION MMUName(mmuID: LONGINT): Str255;BEGIN  CASE mmuID OF    0: MMUName := 'No MMU';    1: MMUName := '68851-style MMU';    2: MMUName := '68030 MMU';    3: MMUName := '68040 MMU';  OTHERWISE    MMUName := CONCAT('Unknown MMU (', CONCAT(LongToStr(mmuID), ')'));  END;END;FUNCTION FPUName(fpuID: LONGINT): Str255;BEGIN  CASE fpuID OF    0: FPUName := 'No FPU';    1: FPUName := '68881';    2: FPUName := '68882';    3: FPUName := '68040 built-in FPU';  OTHERWISE    FPUName := CONCAT('Unknown FPU (', CONCAT(LongToStr(fpuID), ')'));  END;END;FUNCTION QuickDrawName(qdVersion: LONGINT): Str255;BEGIN  CASE qdVersion OF    $0000: QuickDrawName := 'Original 1-bit QuickDraw';    $0100: QuickDrawName := '8-bit Color QuickDraw';    $0200: QuickDrawName := '32-Bit QuickDraw 1.0';    $0210: QuickDrawName := '32-Bit QuickDraw 1.1';    $0220: QuickDrawName := '32-Bit QuickDraw 1.2';    $0230: QuickDrawName := '32-Bit QuickDraw 1.3';  OTHERWISE    QuickDrawName := CONCAT('QuickDraw rev ', HexWord(qdVersion));  END;END;FUNCTION KeyboardName(keyID: LONGINT): Str255;BEGIN  CASE keyID OF    1: KeyboardName := 'Macintosh keyboard';    2: KeyboardName := 'Macintosh keypad';    3: KeyboardName := 'Macintosh Plus keyboard';    4: KeyboardName := 'Extended ADB keyboard';    5: KeyboardName := 'Standard ADB keyboard';    6: KeyboardName := 'Portable keyboard';    7: KeyboardName := 'Portable keyboard (ISO)';    8: KeyboardName := 'Standard ADB keyboard (ISO)';    9: KeyboardName := 'Extended ADB keyboard (ISO)';    10: KeyboardName := 'Apple Keyboard II';    11: KeyboardName := 'Apple Keyboard II (ISO)';    12: KeyboardName := 'PowerBook keyboard';    13: KeyboardName := 'PowerBook keyboard (ISO)';    14: KeyboardName := 'Apple Adjustable keypad';    15: KeyboardName := 'Apple Adjustable keyboard';  OTHERWISE    KeyboardName := CONCAT('Unknown keyboard (', CONCAT(LongToStr(keyID), ')'));  END;END;FUNCTION MachineName(machineID: LONGINT): Str255;BEGIN  CASE machineID OF    1: MachineName := 'Macintosh 128K / 512K family';    2: MachineName := 'Macintosh XL';    3: MachineName := 'Macintosh 512Ke';    4: MachineName := 'Macintosh Plus';    5: MachineName := 'Macintosh SE';    6: MachineName := 'Macintosh II';    7: MachineName := 'Macintosh IIx';    8: MachineName := 'Macintosh IIcx';    9: MachineName := 'Macintosh SE/30';    10: MachineName := 'Macintosh Portable';    11: MachineName := 'Macintosh IIci';    13: MachineName := 'Macintosh IIfx';    17: MachineName := 'Macintosh Classic';    18: MachineName := 'Macintosh IIsi';    19: MachineName := 'Macintosh LC';    20: MachineName := 'Macintosh Quadra 900';    21: MachineName := 'PowerBook 170';    22: MachineName := 'Macintosh Quadra 700';    23: MachineName := 'Macintosh Classic II';    24: MachineName := 'PowerBook 100';    25: MachineName := 'PowerBook 140';  OTHERWISE    MachineName := CONCAT('Unknown machine (', CONCAT(LongToStr(machineID), ')'));  END;END;FUNCTION HiWordOf(value: LONGINT): INTEGER;BEGIN  HiWordOf := value DIV 65536;END;FUNCTION LoWordOf(value: LONGINT): INTEGER;BEGIN  LoWordOf := value MOD 65536;END;FUNCTION PanelTitle(panel: INTEGER): Str255;BEGIN  CASE panel OF    kPanelOverview: PanelTitle := 'Overview';    kPanelMemory:   PanelTitle := 'Memory';    kPanelDisplay:  PanelTitle := 'Display';    kPanelRaw:      PanelTitle := 'Raw IDs';    kPanelAbout:    PanelTitle := 'About';  OTHERWISE    PanelTitle := 'Panel';  END;END;PROCEDURE AddLine(s: Str255);BEGIN  IF gLineCount < kMaxLines THEN  BEGIN    gLines[gLineCount] := s;    gLineCount := gLineCount + 1;  END;END;PROCEDURE AddBlankLine;BEGIN  AddLine('');END;PROCEDURE AddKeyValue(keyName, value: Str255);VAR  s: Str255;BEGIN  s := keyName;  s := CONCAT(s, ': ');  s := CONCAT(s, value);  AddLine(s);END;PROCEDURE AddRawSelector(selector: OSType; caption: Str255);VAR  response: LONGINT;  err: OSErr;  s: Str255;BEGIN  err := Gestalt(selector, response);  IF err = noErr THEN  BEGIN    s := LongToStr(response);    s := CONCAT(s, ' (');    s := CONCAT(s, HexLong(response));    s := CONCAT(s, ')');    AddKeyValue(caption, s);  END  ELSE  BEGIN    s := CONCAT('Unavailable (err ', LongToStr(err));    s := CONCAT(s, ')');    AddKeyValue(caption, s);  END;END;FUNCTION CurrentVolumeName: Str255;BEGIN  CurrentVolumeName := '(default volume)';END;FUNCTION CurrentVRefNum: INTEGER;BEGIN  CurrentVRefNum := 0;END;PROCEDURE RefreshEnvironment;BEGIN  gSysEnv.machineType := 0;  gSysEnv.systemVersion := $0608;  gSysEnv.processor := 1;  gSysEnv.hasFPU := FALSE;  gSysEnv.hasColorQD := FALSE;  gSysEnv.keyBoardType := 0;  gSysEnv.sysVRefNum := 0;  gGestaltAvailable := FALSE;END;PROCEDURE BuildOverviewPanel;VAR  response: LONGINT;  defaultMachineID: LONGINT;  s: Str255;BEGIN  AddLine('This panel combines SysEnvirons data with newer Gestalt calls.');  AddBlankLine;  defaultMachineID := gSysEnv.machineType + 2;  IF Gestalt(gestaltMachineType, response) = noErr THEN    defaultMachineID := response;  AddKeyValue('Machine', MachineName(defaultMachineID));  AddKeyValue('CPU', ProcessorName(gSysEnv.processor));  AddKeyValue('System version', VersionWordToStr(gSysEnv.systemVersion));  AddKeyValue('Gestalt available', BoolToStr(gGestaltAvailable));  AddKeyValue('Color QuickDraw', BoolToStr(gSysEnv.hasColorQD));  AddKeyValue('FPU present', BoolToStr(gSysEnv.hasFPU));  AddKeyValue('Keyboard', KeyboardName(gSysEnv.keyBoardType));  AddKeyValue('AppleTalk driver version', 'Unavailable');  AddKeyValue('System file vRefNum', LongToStr(gSysEnv.sysVRefNum));  AddKeyValue('Current volume', CurrentVolumeName);  AddKeyValue('Current volume refNum', LongToStr(CurrentVRefNum));  IF Gestalt(gestaltSystemVersion, response) = noErr THEN    AddKeyValue('Gestalt system version', VersionWordToStr(response));  IF Gestalt(gestaltROMVersion, response) = noErr THEN    AddKeyValue('ROM version', HexLong(response));  IF Gestalt(gestaltROMSize, response) = noErr THEN    AddKeyValue('ROM size', MBytes(response));  IF Gestalt(gestaltLogicalRAMSize, response) = noErr THEN    AddKeyValue('Logical RAM', MBytes(response));  IF Gestalt(gestaltPhysicalRAMSize, response) = noErr THEN    AddKeyValue('Physical RAM', MBytes(response));  AddBlankLine;  s := CONCAT('System has been up for about ', LongToStr(TickCount DIV 3600));  s := CONCAT(s, ' ');  s := CONCAT(s, Plural('minute', TickCount DIV 3600));  AddKeyValue('Approximate uptime', s);END;PROCEDURE BuildMemoryPanel;VAR  totalPurgeable: LONGINT;  contigPurgeable: LONGINT;  response: LONGINT;BEGIN  AddLine('Classic Memory Manager statistics for the current boot session.');  AddBlankLine;  AddKeyValue('FreeMem', 'Unavailable');  AddKeyValue('MaxMem', 'Unavailable');  AddKeyValue('CompactMem(0)', 'Unavailable');  AddKeyValue('PurgeSpace total', 'Unavailable');  AddKeyValue('PurgeSpace contiguous', 'Unavailable');  IF Gestalt(gestaltLogicalRAMSize, response) = noErr THEN    AddKeyValue('Gestalt logical RAM', MBytes(response));  IF Gestalt(gestaltPhysicalRAMSize, response) = noErr THEN    AddKeyValue('Gestalt physical RAM', MBytes(response));  IF Gestalt(gestaltMMUType, response) = noErr THEN    AddKeyValue('MMU', MMUName(response));  IF Gestalt(gestaltFPUType, response) = noErr THEN    AddKeyValue('FPU', FPUName(response));  AddBlankLine;  AddLine('Tip: refresh after quitting other apps or desk accessories to watch memory change.');END;PROCEDURE BuildDisplayPanel;VAR  qdVersion: LONGINT;  screenWidth: INTEGER;  screenHeight: INTEGER;  rowBytes: INTEGER;  desktopHeight: INTEGER;BEGIN  screenWidth := screenBits.bounds.right - screenBits.bounds.left;  screenHeight := screenBits.bounds.bottom - screenBits.bounds.top;  rowBytes := screenBits.rowBytes MOD 16384;  desktopHeight := screenHeight - 20;  AddLine('Screen and graphics information from QuickDraw globals.');  AddBlankLine;  AddKeyValue('Screen bounds', RectToStr(screenBits.bounds));  AddKeyValue('Screen size',    CONCAT(CONCAT(LongToStr(screenWidth), ' x '), CONCAT(LongToStr(screenHeight), ' pixels')));  AddKeyValue('Usable desktop height', CONCAT(LongToStr(desktopHeight), ' pixels'));  AddKeyValue('Menu bar height', '20 pixels');  AddKeyValue('Row bytes', LongToStr(rowBytes));  AddKeyValue('Color QuickDraw', BoolToStr(gSysEnv.hasColorQD));  IF Gestalt(gestaltQuickdrawVersion, qdVersion) = noErr THEN  BEGIN    AddKeyValue('QuickDraw version', QuickDrawName(qdVersion));    AddKeyValue('QuickDraw raw', HexWord(qdVersion));  END;  AddBlankLine;  AddLine('On monochrome Macs this pane still reports real screen geometry.');END;PROCEDURE BuildRawPanel;VAR  envMachine: LONGINT;BEGIN  AddLine('Raw values are useful when comparing real hardware, ROM revisions, or emulators.');  AddBlankLine;  envMachine := gSysEnv.machineType + 2;  AddKeyValue('SysEnvirons machine', LongToStr(gSysEnv.machineType));  AddKeyValue('Adjusted machine guess', LongToStr(envMachine));  AddKeyValue('SysEnvirons processor', LongToStr(gSysEnv.processor));  AddKeyValue('SysEnvirons keyboard', LongToStr(gSysEnv.keyBoardType));  AddKeyValue('SysEnvirons systemVersion', HexWord(gSysEnv.systemVersion));  AddKeyValue('SysEnvirons sysVRefNum', LongToStr(gSysEnv.sysVRefNum));  AddKeyValue('SysEnvirons AppleTalk', 'Unavailable');  AddBlankLine;  AddRawSelector(gestaltMachineType, 'Gestalt mach');  AddRawSelector(gestaltProcessorType, 'Gestalt proc');  AddRawSelector(gestaltKeyboardType, 'Gestalt kbd ');  AddRawSelector(gestaltSystemVersion, 'Gestalt sysv');  AddRawSelector(gestaltQuickdrawVersion, 'Gestalt qd  ');  AddRawSelector(gestaltROMVersion, 'Gestalt romv');  AddRawSelector(gestaltROMSize, 'Gestalt rom ');  AddRawSelector(gestaltLogicalRAMSize, 'Gestalt lram');  AddRawSelector(gestaltPhysicalRAMSize, 'Gestalt ram ');  AddRawSelector(gestaltMMUType, 'Gestalt mmu ');  AddRawSelector(gestaltFPUType, 'Gestalt fpu ');END;PROCEDURE BuildAboutPanel;BEGIN  AddLine('System Probe is a small resourceless Macintosh Pascal utility.');  AddBlankLine;  AddLine('Designed for System 6.0.8-era Macs and emulators.');  AddLine('UI features:');  AddLine('  - Click the left-side radio buttons to switch panels.');  AddLine('  - Use the View menu for the same panel changes.');  AddLine('  - Use the scrollbar to move through long reports.');  AddLine('  - Choose File > Refresh or press Command-R to rescan.');  AddLine('  - Choose File > Quit or press Command-Q to leave.');  AddBlankLine;  AddLine('Some Gestalt selectors may be unavailable on older ROMs or stripped-down emulators.');  AddLine('When that happens this utility keeps running and reports the error instead.');END;PROCEDURE BuildCurrentPanel;BEGIN  gLineCount := 0;  CASE gCurrentPanel OF    kPanelOverview: BuildOverviewPanel;    kPanelMemory:   BuildMemoryPanel;    kPanelDisplay:  BuildDisplayPanel;    kPanelRaw:      BuildRawPanel;    kPanelAbout:    BuildAboutPanel;  END;END;PROCEDURE SyncViewMenu;VAR  i: INTEGER;BEGIN  FOR i := 1 TO kPanelCount DO    CheckItem(gViewMenu, i, i = gCurrentPanel);END;PROCEDURE SyncScrollBar;VAR  maxValue: INTEGER;BEGIN  maxValue := gLineCount - gVisibleLines;  IF maxValue < 0 THEN    maxValue := 0;  IF gTopLine > maxValue THEN    gTopLine := maxValue;  SetCtlMax(gScrollBar, maxValue);  SetCtlValue(gScrollBar, gTopLine);END;PROCEDURE DrawContent;VAR  i: INTEGER;  y: INTEGER;  lastLine: INTEGER;  titleRect: Rect;BEGIN  SetPort(gMainWindow);  EraseRect(gContentRect);  FrameRect(gContentRect);  titleRect := gContentRect;  InsetRect(titleRect, 8, 6);  TextFont(monaco);  TextSize(9);  TextFace([bold]);  MoveTo(titleRect.left, titleRect.top + 10);  DrawString(PanelTitle(gCurrentPanel));  TextFace([]);  y := titleRect.top + 26;  lastLine := MinI(gLineCount - 1, gTopLine + gVisibleLines - 1);  FOR i := gTopLine TO lastLine DO  BEGIN    MoveTo(titleRect.left, y);    DrawString(gLines[i]);    y := y + kLineHeight;  END;END;PROCEDURE RedrawWindow;BEGIN  SetPort(gMainWindow);  DrawControls(gMainWindow);  DrawContent;END;PROCEDURE RebuildAndRedraw;BEGIN  RefreshEnvironment;  BuildCurrentPanel;  SyncViewMenu;  SyncScrollBar;  RedrawWindow;END;PROCEDURE SelectPanel(panel: INTEGER);VAR  i: INTEGER;BEGIN  gCurrentPanel := panel;  FOR i := kFirstPanel TO kLastPanel DO  BEGIN    IF i = panel THEN      SetCtlValue(gNavButtons[i], 1)    ELSE      SetCtlValue(gNavButtons[i], 0);  END;  gTopLine := 0;  BuildCurrentPanel;  SyncViewMenu;  SyncScrollBar;  DrawContent;END;PROCEDURE ScrollTo(newTop: INTEGER);VAR  maxValue: INTEGER;BEGIN  maxValue := GetCtlMax(gScrollBar);  newTop := MaxI(0, MinI(newTop, maxValue));  IF newTop <> gTopLine THEN  BEGIN    gTopLine := newTop;    SetCtlValue(gScrollBar, gTopLine);    DrawContent;  END;END;PROCEDURE HandleMenuChoice(choice: LONGINT);VAR  theMenu: INTEGER;  theItem: INTEGER;BEGIN  theMenu := HiWordOf(choice);  theItem := LoWordOf(choice);  CASE theMenu OF    kAppleMenuID:      BEGIN        SelectPanel(kPanelAbout);      END;    kFileMenuID:      CASE theItem OF        kCmdRefresh: RebuildAndRedraw;        kCmdQuit: gDone := TRUE;      END;    kViewMenuID:      IF (theItem >= kFirstPanel) AND (theItem <= kLastPanel) THEN        SelectPanel(theItem);  END;  HiliteMenu(0);END;PROCEDURE HandleControlClick(localPt: Point);VAR  whichControl: ControlHandle;  part: INTEGER;  i: INTEGER;BEGIN  part := FindControl(localPt, gMainWindow, whichControl);  IF whichControl = NIL THEN    Exit(HandleControlClick);  IF whichControl = gScrollBar THEN  BEGIN    CASE part OF      inUpButton:   ScrollTo(gTopLine - 1);      inDownButton: ScrollTo(gTopLine + 1);      inPageUp:     ScrollTo(gTopLine - gVisibleLines);      inPageDown:   ScrollTo(gTopLine + gVisibleLines);      inThumb:        BEGIN          ScrollTo(GetCtlValue(gScrollBar));        END;    END;    Exit(HandleControlClick);  END;  FOR i := kFirstPanel TO kLastPanel DO    IF whichControl = gNavButtons[i] THEN    BEGIN      SelectPanel(i);      Exit(HandleControlClick);    END;END;PROCEDURE HandleMouseDown(event: EventRecord);VAR  theWindow: WindowPtr;  windowPart: INTEGER;  localPt: Point;BEGIN  windowPart := FindWindow(event.where, theWindow);  CASE windowPart OF    inMenuBar:      HandleMenuChoice(MenuSelect(event.where));    inSysWindow:      SystemClick(event, theWindow);    inDrag:      DragWindow(theWindow, event.where, screenBits.bounds);    inGoAway:      IF TrackGoAway(theWindow, event.where) THEN        gDone := TRUE;    inContent:      BEGIN        IF theWindow <> FrontWindow THEN          SelectWindow(theWindow)        ELSE        BEGIN          localPt := event.where;          GlobalToLocal(localPt);          HandleControlClick(localPt);        END;      END;  END;END;PROCEDURE HandleKeyDown(event: EventRecord);VAR  ch: CHAR;  choice: LONGINT;BEGIN  ch := CHR(event.message MOD 256);  IF ((event.modifiers DIV cmdKey) MOD 2) <> 0 THEN  BEGIN    choice := MenuKey(ch);    IF choice <> 0 THEN      HandleMenuChoice(choice);    Exit(HandleKeyDown);  END;  CASE ORD(ch) OF    30: ScrollTo(gTopLine - 1);    31: ScrollTo(gTopLine + 1);  END;END;PROCEDURE UpdateWindow(theWindow: WindowPtr);BEGIN  BeginUpdate(theWindow);  IF theWindow = gMainWindow THEN    RedrawWindow;  EndUpdate(theWindow);END;PROCEDURE BuildMenus;VAR  appleTitle: Str255;BEGIN  gAppleMenu := GetMenu(kAppleMenuID);  IF gAppleMenu = NIL THEN  BEGIN    appleTitle := CHR(20);    gAppleMenu := NewMenu(kAppleMenuID, appleTitle);    AppendMenu(gAppleMenu, 'About System Probe...');  END;  AddResMenu(gAppleMenu, 'DRVR');  InsertMenu(gAppleMenu, 0);  gFileMenu := GetMenu(kFileMenuID);  IF gFileMenu = NIL THEN  BEGIN    gFileMenu := NewMenu(kFileMenuID, 'File');    AppendMenu(gFileMenu, 'Refresh/R;Quit/Q');  END;  InsertMenu(gFileMenu, 0);  gViewMenu := GetMenu(kViewMenuID);  IF gViewMenu = NIL THEN  BEGIN    gViewMenu := NewMenu(kViewMenuID, 'View');    AppendMenu(gViewMenu, 'Overview;Memory;Display;Raw IDs;About');  END;  InsertMenu(gViewMenu, 0);  DrawMenuBar;END;PROCEDURE BuildWindow;VAR  bounds: Rect;  buttonRect: Rect;  scrollRect: Rect;  i: INTEGER;BEGIN  SetRect(bounds, 40, 50, 520, 332);  gMainWindow := NewWindow(NIL, bounds, 'System Probe', TRUE, documentProc, WindowPtr(-1), TRUE, 0);  SetPort(gMainWindow);  SetRect(gContentRect, 126, 12, 456, 268);  gVisibleLines := (gContentRect.bottom - gContentRect.top - 34) DIV kLineHeight;  IF gVisibleLines < 1 THEN    gVisibleLines := 1;  SetRect(buttonRect, 14, 16, 112, 36);  FOR i := kFirstPanel TO kLastPanel DO  BEGIN    gNavButtons[i] := NewControl(gMainWindow, buttonRect, PanelTitle(i), TRUE, 0, 0, 1, radioButProc, i);    OffsetRect(buttonRect, 0, 30);  END;  SetRect(scrollRect, gContentRect.right + 4, gContentRect.top, gContentRect.right + 20, gContentRect.bottom);  gScrollBar := NewControl(gMainWindow, scrollRect, '', TRUE, 0, 0, 0, scrollBarProc, 0);  ShowWindow(gMainWindow);  SelectWindow(gMainWindow);END;PROCEDURE Initialize;BEGIN  InitGraf(@thePort);  InitFonts;  InitWindows;  InitMenus;  TEInit;  InitDialogs(NIL);  InitCursor;  gDone := FALSE;  gCurrentPanel := kPanelOverview;  gTopLine := 0;  BuildMenus;  BuildWindow;  RebuildAndRedraw;  SelectPanel(kPanelOverview);END;PROCEDURE EventLoop;VAR  event: EventRecord;BEGIN  REPEAT    IF GetNextEvent(everyEvent, event) THEN    BEGIN      CASE event.what OF        mouseDown: HandleMouseDown(event);        keyDown,        autoKey:   HandleKeyDown(event);        updateEvt: UpdateWindow(WindowPtr(event.message));      END;    END    ELSE      SystemTask;  UNTIL gDone;END;BEGIN  Initialize;  EventLoop;END.
